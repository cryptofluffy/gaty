// Erweiterte Routing Engine für dynamisches Service Management / Advanced Routing Engine for dynamic Service Management
const { logger } = require('./utils');
 
// Route Registry - Dynamische Route Verwaltung / Route Registry - Dynamic Route Management
class RouteRegistry {
  constructor() {
    // Aktive Routes speichern / Store active routes
    this.routes = new Map();
    // Service Health Status / Service health status
    this.serviceHealth = new Map();
    // Load Balancing Strategien / Load balancing strategies
    this.loadBalancers = new Map();
  }
 
  // Route registrieren / Register route
  registerRoute(path, config) {
    // Prüfen ob Route bereits existiert / Check if route already exists
    if (this.routes.has(path)) {
      throw new Error(`Route bereits registriert / Route already registered: ${path}`);
    }
 
    // Route Konfiguration validieren / Validate route configuration
    if (!this.validateRouteConfig(config)) {
      throw new Error(`Ungültige Route Konfiguration für ${path} / Invalid route configuration for ${path}`);
    }
 
    // Route mit Metadaten speichern / Store route with metadata
    this.routes.set(path, {
      ...config,
      createdAt: new Date().toISOString(), // Erstellungszeitpunkt / Creation timestamp
      lastAccessed: null, // Letzter Zugriff / Last access
      requestCount: 0, // Anfragen-Zähler / Request counter
      errors: 0 // Fehler-Zähler / Error counter
    });
 
    logger.info(`Route registriert / Route registered: ${path} -> ${JSON.stringify(config.targets)}`);
  }
 
  // Route entfernen / Remove route
  unregisterRoute(path) {
    if (this.routes.delete(path)) {
      logger.info(`Route entfernt / Route removed: ${path}`);
      return true;
    }
    return false;
  }
 
  // Beste Route für Request finden / Find best route for request
  findRoute(requestPath, method = 'GET') {
    // Exakte Übereinstimmung suchen und HTTP Methode prüfen / Look for exact match and check HTTP method
    if (this.routes.has(requestPath)) {
      const routeConfig = this.routes.get(requestPath);
      // HTTP Methoden prüfen / Check HTTP methods
      if (routeConfig.methods && !routeConfig.methods.includes(method)) {
        return null;
      }
      return this.updateRouteStats(requestPath);
    }
 
    // Wildcard und Parameter Routen suchen / Search for wildcard and parameter routes
    for (const [routePath, config] of this.routes) {
      if (this.matchRoute(routePath, requestPath, method)) {
        return this.updateRouteStats(routePath);
      }
    }
 
    return null; // Keine Route gefunden / No route found
  }
 
  // Route Pattern Matching / Route pattern matching
  matchRoute(routePath, requestPath, method) {
    // Route Konfiguration abrufen / Get route configuration
    const routeConfig = this.routes.get(routePath);
    
    // HTTP Methoden prüfen / Check HTTP methods
    if (routeConfig && routeConfig.methods && !routeConfig.methods.includes(method)) {
      return false;
    }
 
    // Wildcard Unterstützung / Wildcard support
    if (routePath.includes('*')) {
      const pattern = routePath.replace(/\*/g, '.*');
      return new RegExp(`^${pattern}$`).test(requestPath);
    }
 
    // Parameter Routen (:id, :name, etc.) / Parameter routes
    if (routePath.includes(':')) {
      const pattern = routePath.replace(/:[\w]+/g, '[\\w-]+');
      return new RegExp(`^${pattern}$`).test(requestPath);
    }
 
    return routePath === requestPath;
  }
 
  // Route Statistiken aktualisieren / Update route statistics
  updateRouteStats(routePath) {
    const route = this.routes.get(routePath);
    if (route) {
      route.lastAccessed = new Date().toISOString(); // Letzter Zugriff aktualisieren / Update last access
      route.requestCount++; // Anfragen-Zähler erhöhen / Increment request counter
      this.routes.set(routePath, route);
    }
    return route;
  }
 
  // Route Konfiguration validieren / Validate route configuration
  validateRouteConfig(config) {
    // Pflichtfelder prüfen / Check required fields
    if (!config.targets || !Array.isArray(config.targets) || config.targets.length === 0) {
      return false;
    }
 
    // Target URLs validieren / Validate target URLs
    return config.targets.every(target => {
      return target.url && typeof target.url === 'string' && target.url.startsWith('http');
    });
  }
 
  // Alle registrierten Routen abrufen / Get all registered routes
  getAllRoutes() {
    return Array.from(this.routes.entries()).map(([path, config]) => ({
      path,
      ...config
    }));
  }
}
 
// Load Balancer Strategien / Load Balancer Strategies
class LoadBalancer {
  constructor(strategy = 'round-robin') {
    this.strategy = strategy;
    this.currentIndex = new Map(); // Index für Round-Robin / Index for round-robin
    this.weights = new Map(); // Gewichtungen für Weighted Round-Robin / Weights for weighted round-robin
  }
 
  // Nächstes Target basierend auf Strategie auswählen / Select next target based on strategy
  selectTarget(targets, routePath) {
    if (!targets || targets.length === 0) {
      return null;
    }
 
    // Nur gesunde Targets verwenden / Only use healthy targets
    const healthyTargets = targets.filter(target => this.isTargetHealthy(target));
    if (healthyTargets.length === 0) {
      logger.warn(`Keine gesunden Targets für Route ${routePath} / No healthy targets for route ${routePath}`);
      return targets[0]; // Fallback zum ersten Target / Fallback to first target
    }
 
    switch (this.strategy) {
      case 'round-robin':
        return this.roundRobin(healthyTargets, routePath);
      
      case 'least-connections':
        return this.leastConnections(healthyTargets);
      
      case 'weighted':
        return this.weighted(healthyTargets, routePath);
      
      case 'random':
        return this.random(healthyTargets);
      
      default:
        return healthyTargets[0]; // Standard: Erstes Target / Default: First target
    }
  }
 
  // Round-Robin Load Balancing / Round-robin load balancing
  roundRobin(targets, routePath) {
    const currentIdx = this.currentIndex.get(routePath) || 0;
    const nextIdx = (currentIdx + 1) % targets.length;
    this.currentIndex.set(routePath, nextIdx);
    
    logger.debug(`Round-Robin: Target ${nextIdx + 1}/${targets.length} für Route ${routePath}`);
    return targets[nextIdx];
  }
 
  // Least Connections Load Balancing / Least connections load balancing
  leastConnections(targets) {
    // Target mit geringsten aktiven Verbindungen finden / Find target with least active connections
    let leastTarget = targets[0];
    let leastConnections = leastTarget.activeConnections || 0;
 
    for (const target of targets) {
      const connections = target.activeConnections || 0;
      if (connections < leastConnections) {
        leastConnections = connections;
        leastTarget = target;
      }
    }
 
    logger.debug(`Least Connections: Target mit ${leastConnections} Verbindungen ausgewählt`);
    return leastTarget;
  }
 
  // Weighted Load Balancing / Weighted load balancing
  weighted(targets, routePath) {
    // Gewichtungen berücksichtigen / Consider weights
    const totalWeight = targets.reduce((sum, target) => sum + (target.weight || 1), 0);
    const random = Math.random() * totalWeight;
    
    let currentWeight = 0;
    for (const target of targets) {
      currentWeight += target.weight || 1;
      if (random <= currentWeight) {
        logger.debug(`Weighted: Target mit Gewicht ${target.weight || 1} ausgewählt`);
        return target;
      }
    }
 
    return targets[0]; // Fallback / Fallback
  }
 
  // Random Load Balancing / Random load balancing
  random(targets) {
    const randomIndex = Math.floor(Math.random() * targets.length);
    logger.debug(`Random: Target ${randomIndex + 1}/${targets.length} ausgewählt`);
    return targets[randomIndex];
  }
 
  // Target Gesundheitsstatus prüfen / Check target health status
  isTargetHealthy(target) {
    // Health Check Status prüfen / Check health check status
    return target.healthy !== false && !target.disabled;
  }
}
 
// Routing Middleware für Express / Routing Middleware for Express
const createRoutingMiddleware = (routeRegistry, loadBalancer) => {
  return (req, res, next) => {
    // Route für aktuellen Request finden / Find route for current request
    const route = routeRegistry.findRoute(req.path, req.method);
    
    if (!route) {
      // Keine Route gefunden, weiter zur nächsten Middleware / No route found, continue to next middleware
      return next();
    }
 
    // Target für Load Balancing auswählen / Select target for load balancing
    const target = loadBalancer.selectTarget(route.targets, req.path);
    
    if (!target || !loadBalancer.isTargetHealthy(target)) {
      // Kein verfügbares oder ungesundes Target / No available or unhealthy target
      return res.status(503).json({
        error: 'Service nicht verfügbar / Service unavailable',
        message: 'Keine gesunden Backend-Services verfügbar / No healthy backend services available'
      });
    }
 
    // Target Information an Request anhängen / Attach target information to request
    req.routingTarget = target;
    req.routeConfig = route;
    
    logger.info(`Request geroutet / Request routed: ${req.method} ${req.path} -> ${target.url}`);
    next();
  };
};
