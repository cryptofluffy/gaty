const { SecurityManager } = require('../security');
 
describe('SecurityManager', () => {
  let securityManager;
 
  beforeEach(() => {
    securityManager = new SecurityManager({
      jwtSecret: 'test-secret-key-32-characters-long-for-security',
      saltRounds: 10 // Faster for tests
    });
  });
 
  afterEach(() => {
    if (securityManager.destroy) {
      securityManager.destroy();
    }
  });
 
  describe('Initialization', () => {
    test('should initialize with valid JWT secret', () => {
      expect(securityManager.config.jwtSecret).toBe('test-secret-key-32-characters-long-for-security');
    });
 
    test('should throw error for short JWT secret', () => {
      expect(() => {
        new SecurityManager({ jwtSecret: 'short' });
      }).toThrow('JWT Secret must be at least 32 characters long for security');
    });
 
    test('should generate secure key when no JWT secret provided', () => {
      // Mock environment variable
      delete process.env.JWT_SECRET;
      
      const manager = new SecurityManager();
      expect(manager.config.jwtSecret.length).toBeGreaterThanOrEqual(32);
      
      if (manager.destroy) {
        manager.destroy();
      }
    });
  });
 
  describe('Password Management', () => {
    test('should hash password correctly', async () => {
      const password = 'TestPassword123!';
      const hash = await securityManager.hashPassword(password);
      
      expect(hash).toBeDefined();
      expect(hash).not.toBe(password);
      expect(hash.length).toBeGreaterThan(50); // bcrypt hashes are long
    });
 
    test('should verify password correctly', async () => {
      const password = 'TestPassword123!';
      const hash = await securityManager.hashPassword(password);
      
      const isValid = await securityManager.verifyPassword(password, hash);
      expect(isValid).toBe(true);
      
      const isInvalid = await securityManager.verifyPassword('wrongpassword', hash);
      expect(isInvalid).toBe(false);
    });
 
    test('should reject weak passwords', async () => {
      const weakPasswords = [
        'weak',
        'password',
        '123456',
        'qwerty',
        'admin',
        'Test1' // Too short
      ];
 
      for (const password of weakPasswords) {
        await expect(
          securityManager.hashPassword(password)
        ).rejects.toThrow(/Passwort zu schwach|Password too weak/);
      }
    });
 
    test('should accept strong passwords', async () => {
      const strongPasswords = [
        'StrongPassword123!',
        'MySecure#Pass2024',
        'Complex$Password99'
      ];
 
      for (const password of strongPasswords) {
        const hash = await securityManager.hashPassword(password);
        expect(hash).toBeDefined();
      }
    });
  });
 
  describe('Password Strength Checking', () => {
    test('should correctly assess password strength', () => {
      const tests = [
        { password: 'weak', expectedStrength: 'weak' },
        { password: 'StrongPass1!', expectedStrength: 'strong' },
        { password: 'MediumPass1', expectedStrength: 'medium' },
        { password: 'password', expectedStrength: 'weak' } // Common password
      ];
 
      tests.forEach(({ password, expectedStrength }) => {
        const result = securityManager.checkPasswordStrength(password);
        expect(result.strength).toBe(expectedStrength);
      });
    });
 
    test('should provide helpful feedback for weak passwords', () => {
      const result = securityManager.checkPasswordStrength('weak');
      
      expect(result.feedback).toContain('Mindestens 8 Zeichen erforderlich');
      expect(result.feedback).toContain('Großbuchstaben erforderlich');
      expect(result.feedback).toContain('Zahlen erforderlich');
      expect(result.feedback).toContain('Sonderzeichen erforderlich');
    });
  });
 
  describe('Encryption/Decryption', () => {
    test('should encrypt and decrypt data correctly', () => {
      const plaintext = 'This is secret data';
      
      const encrypted = securityManager.encrypt(plaintext);
      expect(encrypted).toHaveProperty('encrypted');
      expect(encrypted).toHaveProperty('iv');
      expect(encrypted).toHaveProperty('authTag');
      
      const decrypted = securityManager.decrypt(encrypted);
      expect(decrypted).toBe(plaintext);
    });
 
    test('should use custom encryption key', () => {
      const plaintext = 'Custom key test';
      const customKey = 'custom-key-32-chars-long-for-test';
      
      const encrypted = securityManager.encrypt(plaintext, customKey);
      const decrypted = securityManager.decrypt(encrypted, customKey);
      
      expect(decrypted).toBe(plaintext);
    });
 
    test('should fail to decrypt with wrong key', () => {
      const plaintext = 'Secret message';
      const encrypted = securityManager.encrypt(plaintext);
      
      expect(() => {
        securityManager.decrypt(encrypted, 'wrong-key-32-chars-long-for-test');
      }).toThrow();
    });
 
    test('should fail to decrypt tampered data', () => {
      const plaintext = 'Secret message';
      const encrypted = securityManager.encrypt(plaintext);
      
      // Tamper with the encrypted data
      encrypted.encrypted = encrypted.encrypted.substring(0, -4) + 'xxxx';
      
      expect(() => {
        securityManager.decrypt(encrypted);
      }).toThrow();
    });
  });
 
  describe('API Key Generation', () => {
    test('should generate secure API key', () => {
      const apiKey = securityManager.generateSecureKey(32);
      
      expect(apiKey).toBeDefined();
      expect(apiKey.length).toBe(64); // 32 bytes = 64 hex chars
      expect(/^[a-f0-9]+$/.test(apiKey)).toBe(true);
    });
 
    test('should generate different keys each time', () => {
      const key1 = securityManager.generateSecureKey();
      const key2 = securityManager.generateSecureKey();
      
      expect(key1).not.toBe(key2);
    });
  });
 
  describe('Security Events', () => {
    test('should log security events', () => {
      securityManager.logSecurityEvent('test_event', { 
        details: 'test event data' 
      });
      
      expect(securityManager.securityEvents).toHaveLength(1);
      expect(securityManager.securityEvents[0].type).toBe('test_event');
      expect(securityManager.securityEvents[0].details).toEqual({ 
        details: 'test event data' 
      });
    });
 
    test('should maintain event history limit', () => {
      // Create manager with small event limit for testing
      const testManager = new SecurityManager({
        jwtSecret: 'test-secret-key-32-characters-long-for-security',
        maxSecurityEvents: 3
      });
 
      // Add more events than the limit
      for (let i = 0; i < 5; i++) {
        testManager.logSecurityEvent('test_event', { index: i });
      }
 
      expect(testManager.securityEvents.length).toBeLessThanOrEqual(3);
      
      if (testManager.destroy) {
        testManager.destroy();
      }
    });
  });
 
  describe('Cleanup and Resource Management', () => {
    test('should clean up resources when destroyed', () => {
      const manager = new SecurityManager({
        jwtSecret: 'test-secret-key-32-characters-long-for-security'
      });
 
      // Add some data
      manager.tokenBlacklist.add('test-token');
      manager.failedAttempts.set('192.168.1.1', 3);
      manager.logSecurityEvent('test', {});
 
      // Destroy and check cleanup
      manager.destroy();
 
      expect(manager.tokenBlacklist.size).toBe(0);
      expect(manager.failedAttempts.size).toBe(0);
      expect(manager.securityEvents.length).toBe(0);
      expect(manager.cleanupTimer).toBeNull();
    });
 
    test('should handle destroy being called multiple times', () => {
      const manager = new SecurityManager({
        jwtSecret: 'test-secret-key-32-characters-long-for-security'
      });
 
      expect(() => {
        manager.destroy();
        manager.destroy(); // Should not throw
      }).not.toThrow();
    });
  });
 
  describe('Error Handling', () => {
    test('should handle encryption errors gracefully', () => {
      expect(() => {
        securityManager.encrypt(null);
      }).toThrow();
    });
 
    test('should handle verification errors gracefully', async () => {
      const result = await securityManager.verifyPassword('test', 'invalid-hash');
      expect(result).toBe(false);
    });
 
    test('should handle malformed encrypted data', () => {
      expect(() => {
        securityManager.decrypt({ encrypted: 'invalid', iv: 'invalid', authTag: 'invalid' });
      }).toThrow();
    });
  });
});
